class Solution
{
    //Function to reverse a linked list.
    Node reverseList(Node head)
    {
        // code here
        Node prev = null;
        Node current = head;
        Node next = current.next;
        
        while(next != null)
        {
            current.next = prev;
            prev = current;
            current = next;
            next = next.next;
        }
        
        current.next = prev;
        prev = current;
        
        return prev;
    }
}
